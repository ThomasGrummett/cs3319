-- P2
CREATE VIEW  PurchaseInfo(FirstName,LastName,Description,Quantity,Cost,TotalCost) AS 
SELECT Customers.FirstName, Customers.LastName,Products.Description, Buys.Quantity,Products.Cost, Buys.Quantity*Products.Cost
FROM Customers 
	INNER JOIN Buys WHERE Customers.CustomerID=Buys.CustomerID
	INNER JOIN Products WHERE Buys.ProductID=Products.ProductID
	ORDER BY LastName;
	
-- P3
SELECT * FROM PurchaseInfo;
-- P4
SELECT * FROM PurchaseInfo WHERE TotalCost>100;
